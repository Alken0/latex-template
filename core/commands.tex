% print pages
\newcommand{\printListOfFigures}{
	\cleardoublepage
	\phantomsection
	\addcontentsline{toc}{chapter}{\listfigurename}
	\listoffigures
}

\newcommand{\printListOfTables}{
	\cleardoublepage
	\phantomsection
	\addcontentsline{toc}{chapter}{\listtablename}
	\listoftables
}
\newcommand{\setPageNumberingToLowercaseRoman}{
	\pagenumbering{roman}
}

\newcommand{\setPageNumberingToArabicStartingFromOne}{
	\cleardoublepage{}
	\pagenumbering{arabic}
	\setcounter{page}{1}
}

\newcommand{\startAppendixIncludingRomanPageNumbering}{
	\cleardoublepage{}
	\appendix
	\pagenumbering{Roman}
}
